# GIT: source control, tracks files and folders
# conpare the prev state and current state of the folder
# Initialize local git
git init

# Add file to the git
git add file_name   //add single files
git add .           //add multiple files

# Commit file to git
git commit =m "message"

# Status check of this repo
git status

# Check logs
git log

# Global user and email/ defining identity
git config --global user.name "your_name"
git config --global ser.email your_email

# Branch
# check bhanches
git branch

# Create new branch
git branch branch_name

# Switch branch
git switch branch_name
git checkout branch_name
